import { createClient } from 'npm:@supabase/supabase-js@2';

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type, x-buildmaster-version',
};

class HttpError extends Error {
  status: number;
  code: string;
  details?: Record<string, unknown>;
  constructor(status: number, code: string, message: string, details?: Record<string, unknown>) {
    super(message);
    this.status = status;
    this.code = code;
    this.details = details;
  }
}

function respond(body: unknown, status = 200) {
  return new Response(JSON.stringify(body), {
    status,
    headers: { ...corsHeaders, 'Content-Type': 'application/json', 'Cache-Control': 'no-store' }
  });
}

function versionParts(value: string) {
  return String(value || '0').replace(/^v/i, '').split(/[.-]/).slice(0, 4).map((part) => Number.parseInt(part.replace(/\D/g, ''), 10) || 0);
}

function compareVersions(left: string, right: string) {
  const a = versionParts(left);
  const b = versionParts(right);
  for (let index = 0; index < Math.max(a.length, b.length, 3); index += 1) {
    const delta = (a[index] || 0) - (b[index] || 0);
    if (delta !== 0) return delta > 0 ? 1 : -1;
  }
  return 0;
}

function base64ToBytes(value: string) {
  const normalized = value.replace(/-/g, '+').replace(/_/g, '/');
  const binary = atob(normalized);
  return Uint8Array.from(binary, (character) => character.charCodeAt(0));
}

function hex(bytes: Uint8Array) {
  return Array.from(bytes, (item) => item.toString(16).padStart(2, '0')).join('');
}

function derEcdsaToRaw(signature: Uint8Array, size = 32) {
  let offset = 0;
  if (signature[offset++] !== 0x30) throw new Error('Assinatura DER inválida.');
  let sequenceLength = signature[offset++];
  if (sequenceLength & 0x80) {
    const lengthBytes = sequenceLength & 0x7f;
    sequenceLength = 0;
    for (let index = 0; index < lengthBytes; index += 1) sequenceLength = (sequenceLength << 8) | signature[offset++];
  }
  if (signature[offset++] !== 0x02) throw new Error('Componente R inválido.');
  let rLength = signature[offset++];
  const r = signature.slice(offset, offset + rLength);
  offset += rLength;
  if (signature[offset++] !== 0x02) throw new Error('Componente S inválido.');
  let sLength = signature[offset++];
  const s = signature.slice(offset, offset + sLength);

  const normalize = (component: Uint8Array) => {
    let clean = component;
    while (clean.length > size && clean[0] === 0) clean = clean.slice(1);
    if (clean.length > size) throw new Error('Componente ECDSA excedeu o tamanho.');
    const output = new Uint8Array(size);
    output.set(clean, size - clean.length);
    return output;
  };
  const raw = new Uint8Array(size * 2);
  raw.set(normalize(r), 0);
  raw.set(normalize(s), size);
  return raw;
}

async function verifyDeviceProof(input: {
  userId: string;
  deviceId: string;
  publicKey: string;
  timestamp: number;
  signature: string;
  appVersion: string;
}) {
  if (!input.deviceId.startsWith('bm2-')) throw new HttpError(403, 'DEVICE_ID_INVALID', 'Identidade segura do aparelho inválida.');
  if (!Number.isFinite(input.timestamp) || Math.abs(Date.now() - input.timestamp) > 5 * 60 * 1000) {
    throw new HttpError(403, 'DEVICE_PROOF_EXPIRED', 'A validação do aparelho expirou. Confira o relógio e tente novamente.');
  }
  const publicKeyBytes = base64ToBytes(input.publicKey);
  const fingerprint = `bm2-${hex(new Uint8Array(await crypto.subtle.digest('SHA-256', publicKeyBytes)))}`;
  if (fingerprint !== input.deviceId) throw new HttpError(403, 'DEVICE_KEY_MISMATCH', 'A chave do aparelho não corresponde ao identificador informado.');
  const key = await crypto.subtle.importKey('spki', publicKeyBytes, { name: 'ECDSA', namedCurve: 'P-256' }, false, ['verify']);
  const message = `${input.deviceId}|${input.timestamp}|${input.appVersion}|${input.userId}`;
  const derSignature = base64ToBytes(input.signature);
  const rawSignature = derEcdsaToRaw(derSignature);
  const valid = await crypto.subtle.verify({ name: 'ECDSA', hash: 'SHA-256' }, key, rawSignature, new TextEncoder().encode(message));
  if (!valid) throw new HttpError(403, 'DEVICE_PROOF_INVALID', 'A assinatura criptográfica do aparelho não confere.');
}

Deno.serve(async (request) => {
  if (request.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });
  try {
    if (request.method !== 'POST') throw new HttpError(405, 'METHOD_NOT_ALLOWED', 'Método não permitido.');
    const authorization = request.headers.get('Authorization');
    if (!authorization) throw new HttpError(401, 'SESSION_MISSING', 'Sessão ausente.');
    const token = authorization.replace(/^Bearer\s+/i, '');
    const supabaseUrl = Deno.env.get('SUPABASE_URL')!;
    const serviceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!;
    const service = createClient(supabaseUrl, serviceKey, { auth: { persistSession: false, autoRefreshToken: false } });
    const { data: authData, error: authError } = await service.auth.getUser(token);
    if (authError || !authData.user) throw new HttpError(401, 'SESSION_INVALID', 'Sua sessão não é mais válida.');

    const body = await request.json().catch(() => ({})) as Record<string, unknown>;
    const appId = String(body.appId || '');
    const appVersion = String(body.appVersion || request.headers.get('X-BuildMaster-Version') || '').trim();
    const { data: settingsRow } = await service.from('buildmaster_security_settings').select('*').eq('id', 1).maybeSingle();
    const settings = settingsRow || {
      min_app_version: '26.75.0', allow_legacy_clients: false, require_device_proof: true,
      admin_mfa_required: true, user_offline_grace_hours: 4, admin_offline_grace_hours: 12
    };

    if (appId !== 'com.buildmaster.elitetatico') throw new HttpError(403, 'APP_ID_INVALID', 'Aplicativo não reconhecido pelo servidor.');
    if (!appVersion && !settings.allow_legacy_clients) {
      throw new HttpError(426, 'LEGACY_CLIENT_BLOCKED', 'Este APK antigo foi bloqueado. Instale a versão mais recente.', { minimumVersion: settings.min_app_version });
    }
    if (appVersion && compareVersions(appVersion, String(settings.min_app_version)) < 0) {
      throw new HttpError(426, 'UPDATE_REQUIRED', `Esta versão foi desativada. Instale o BuildMaster ${settings.min_app_version} ou superior.`, { minimumVersion: settings.min_app_version });
    }

    const { data: profile, error: profileError } = await service.from('buildmaster_profiles').select('*').eq('id', authData.user.id).single();
    if (profileError || !profile) throw new HttpError(403, 'PROFILE_MISSING', 'Perfil da conta não encontrado.');
    if (profile.status === 'blocked') throw new HttpError(403, 'ACCOUNT_BLOCKED', 'Esta conta foi bloqueada pelo administrador.');
    if (profile.status === 'suspended') throw new HttpError(403, 'ACCOUNT_SUSPENDED', 'Esta conta está suspensa.');
    if (profile.expires_at && Date.parse(profile.expires_at) <= Date.now()) throw new HttpError(403, 'ACCOUNT_EXPIRED', 'Seu prazo de acesso terminou.');

    const deviceId = String(body.deviceId || '').trim();
    const devicePublicKey = String(body.devicePublicKey || '').trim();
    const proofTimestamp = Number(body.proofTimestamp || 0);
    const proofSignature = String(body.proofSignature || '').trim();
    const securityVersion = Number(body.securityVersion || 0);
    if (!deviceId) throw new HttpError(403, 'DEVICE_MISSING', 'Identificador do aparelho ausente.');

    if (settings.require_device_proof) {
      if (securityVersion < 2 || !devicePublicKey || !proofSignature) {
        throw new HttpError(426, 'SECURE_DEVICE_REQUIRED', 'Este APK não possui a proteção criptográfica obrigatória. Instale a versão mais recente.', { minimumVersion: settings.min_app_version });
      }
      await verifyDeviceProof({ userId: authData.user.id, deviceId, publicKey: devicePublicKey, timestamp: proofTimestamp, signature: proofSignature, appVersion });
    }

    const { data: deviceRegistration, error: deviceRegistrationError } = await service.rpc('buildmaster_register_secure_device', {
      p_user_id: authData.user.id,
      p_device_id: deviceId,
      p_device_name: String(body.deviceName || 'Aparelho').slice(0, 120),
      p_platform: String(body.platform || 'unknown').slice(0, 240),
      p_public_key: devicePublicKey,
      p_algorithm: String(body.proofAlgorithm || '').slice(0, 40),
      p_security_version: Math.max(2, securityVersion),
      p_proof_at: proofTimestamp,
      p_max_devices: Number(profile.max_devices || 1)
    });
    if (deviceRegistrationError) throw new HttpError(500, 'DEVICE_REGISTRATION_FAILED', deviceRegistrationError.message);
    const registration = (deviceRegistration || {}) as Record<string, unknown>;
    if (!registration.ok) {
      const code = String(registration.code || 'DEVICE_REGISTRATION_FAILED');
      if (code === 'DEVICE_CLONE_BLOCKED') throw new HttpError(403, code, 'A identidade deste aparelho não corresponde ao vínculo original.');
      if (code === 'DEVICE_REPLAY_BLOCKED') throw new HttpError(403, code, 'Uma validação repetida do aparelho foi bloqueada. Tente novamente.');
      if (code === 'DEVICE_LIMIT') throw new HttpError(403, code, 'Limite de aparelhos atingido. Peça ao administrador para desconectar um aparelho.');
      if (code === 'SECURE_DEVICE_REQUIRED') throw new HttpError(426, code, 'Este APK não possui a proteção criptográfica obrigatória.', { minimumVersion: settings.min_app_version });
      throw new HttpError(403, code, 'Não foi possível vincular este aparelho com segurança.');
    }

    const now = new Date().toISOString();
    const offlineGraceHours = profile.role === 'admin'
      ? Number(settings.admin_offline_grace_hours || 12)
      : Number(settings.user_offline_grace_hours || 4);
    await service.from('buildmaster_profiles').update({ last_access_at: now, offline_grace_hours: offlineGraceHours }).eq('id', authData.user.id);

    return respond({
      profile: {
        id: profile.id,
        username: profile.username,
        display_name: profile.display_name,
        role: profile.role,
        status: profile.status,
        plan: profile.plan,
        expires_at: profile.expires_at,
        max_devices: profile.max_devices,
        offline_grace_hours: offlineGraceHours,
        last_access_at: now
      },
      validatedAt: now,
      minimumVersion: settings.min_app_version,
      deviceSecurity: securityVersion >= 2 ? 'keystore-ecdsa' : 'legacy'
    });
  } catch (error) {
    if (error instanceof HttpError) return respond({ error: error.message, code: error.code, ...(error.details || {}) }, error.status);
    return respond({ error: error instanceof Error ? error.message : 'Falha ao validar a licença.', code: 'LICENSE_FAILURE' }, 500);
  }
});
